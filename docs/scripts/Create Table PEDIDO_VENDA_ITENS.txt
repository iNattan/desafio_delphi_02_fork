CREATE TABLE PEDIDO_VENDA_ITENS (
	ID INTEGER NOT NULL,
	ID_PEDIDO_VENDA INTEGER NOT NULL,
	ID_PRODUTO INTEGER NOT NULL,
	QUANTIDADE NUMERIC(15,2) NOT NULL,
	VALOR_UNITARIO NUMERIC(15,2) NOT NULL,
	VALOR_TOTAL NUMERIC(15,2) NOT NULL,
	CONSTRAINT PEDIDO_VENDA_ITENS_PK PRIMARY KEY (ID),
	CONSTRAINT PDV_ITENS_PDV_FK FOREIGN KEY (ID_PEDIDO_VENDA) REFERENCES PEDIDOS_VENDAS(ID),
	CONSTRAINT PDV_ITENS_PRODUTOS_FK FOREIGN KEY (ID_PRODUTO) REFERENCES PRODUTOS(ID)
);
